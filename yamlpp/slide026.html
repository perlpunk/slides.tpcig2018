# YAML.pm, YAML::Tiny $VAR1 = { 'bool false' => 'false', 'bool true' => 'true' };

# YAML::Syck $YAML::Syck::ImplicitTyping = 1; $VAR1 = { 'bool false' => '', 'bool true' => 1 };

# YAML::XS $YAML::XS::Boolean = "JSON::PP"; # since v0.67 $VAR1 = { 'bool false' => bless( do{\(my $o = 0)}, 'JSON::PP::Boolean' ), 'bool true' => bless( do{\(my $o = 1)}, 'JSON::PP::Boolean' ) };

# YAML::PP my $ypp = YAML::PP::Loader->new(boolean => "JSON::PP"); $VAR1 = { 'bool false' => bless( do{\(my $o = 0)}, 'JSON::PP::Boolean' ), 'bool true' => bless( do{\(my $o = 1)}, 'JSON::PP::Boolean' ) };
